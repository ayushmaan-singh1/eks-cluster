pipeline {
    agent any

    stages {
        stage('Clone Repository') {
            steps {
                // Clone the repository from Git
                git branch: 'main', url: 'https://github.com/ayushmaan-singh1/eks-cluster.git'
            }
        }

        stage('Terraform Init') {
            steps {
                // Initialize Terraform
                sh 'terraform init'
            }
        }

        stage('Terraform Plan') {
            steps {
                // Terraform Plan to preview the changes
                sh 'terraform plan"'
            }
        }

        stage('Terraform Apply') {
            steps {
                // Apply the changes to create the EKS cluster
                sh 'terraform apply -auto-approve"'
            }
        }

        stage('Deploy Hello World Application') {
            steps {
                // Deploy the Hello World application to the EKS cluster
                script {
                    sh '''
                    kubectl apply -f hello.yaml
                    kubectl apply -f service.yaml
                    kubectl apply -f ingress.yaml
                    '''
                }
            }
        }
    }

    post {
        always {
            // Clean up workspace after build
            cleanWs()
        }
        success {
            // Notify on success (e.g., send email or Slack message)
            echo 'Deployment succeeded!'
        }
        failure {
            // Notify on failure (e.g., send email or Slack message)
            echo 'Deployment failed!'
        }
    }
}
